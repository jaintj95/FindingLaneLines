import math

def grayscale(img):
    """
    Applies the Grayscale transform
    This will return an image with only one color channel
    but NOTE: to see the returned image as grayscale
    (assuming your grayscaled image is called 'gray')
    you should call plt.imshow(gray, cmap='gray')
    
    Args:
        img: Color image to be converted to grayscale
        
    Returns:
        Grayscale image
    """
    return cv2.cvtColor(img, cv2.COLOR_RGB2GRAY)
    # Or use BGR2GRAY if you read an image with cv2.imread()
    # return cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    

def gaussian_blur(img, kernel_size):
    """
    Applies a Gaussian Noise kernel.
    Gaussian Kernel is a low pass filter kernel 
    which is useful for removing noise.
    Low pass implies lower values pass through 
    while higher values (mostly noise) are blocked.
    
    Args:
        img: Image to be smoothed
        kernel_size: size of the gaussian kernel (should be odd)
        
    Returns:
        Blurred image with gaussian noise removed
    """
    return cv2.GaussianBlur(img, (kernel_size, kernel_size), 0)
    
    
def canny(img, low_threshold, high_threshold):
    """
    Applies the Canny transform
    The algorithm will first detect strong edge (strong gradient) pixels 
    above the high_threshold, and reject pixels below the low_threshold. 
    Next, pixels with values between the low_threshold and high_threshold 
    will be included as long as they are connected to strong edges.
    
    Args:
        img: Image on which edge detection is to be performed
        low_threshold: lower threshold for edge detection
        high_threshold: higher threshold for edge detection
        
    Returns:
        Binary img with white pixels tracing out detected edges & black elsewhere.
    """
    return cv2.Canny(img, low_threshold, high_threshold)


def region_of_interest(img, vertices):
    """
    Applies an image mask.
    
    Only keeps the region of the image defined by the polygon
    formed from `vertices`. The rest of the image is set to black.
    `vertices` should be a numpy array of (array of integer points)
    """
    #defining a blank mask to start with
    mask = np.zeros_like(img)   
    
    #defining a 3 channel or 1 channel color to fill the mask with depending on the input image
    if len(img.shape) > 2:
        channel_count = img.shape[2]  # i.e. 3 or 4 depending on your image
        ignore_mask_color = (255,) * channel_count
    else:
        ignore_mask_color = 255
        
    #filling pixels inside the polygon defined by "vertices" with the fill color    
    cv2.fillPoly(mask, vertices, ignore_mask_color)
    
    #returning the image only where mask pixels are nonzero
    masked_image = cv2.bitwise_and(img, mask)
    return masked_image


def draw_helper(img, X, Y, color, thickness=10):
    """
    Takes in a collection of X and Y points, fits a line to them
    and returns the coordinates of final line to be drawn
    """
    
    # y1 and y2 are fixed because we know lines 
    # should start at bottom and end somewhere in middle
    
    if (len(X) == 0 or len(Y) == 0):
        return
    
    y1 = (img.shape[0] // 2) + 70 
    y2 = img.shape[0] 
    
    
    slope, intercept = np.polyfit(X, Y, deg=1)
    
    # y = mx + b i.e. y = slope*x + b
    x1 = (y1 - intercept)/slope
    x2 = (y2 - intercept)/slope
    
    
    cv2.line(img, (int(x1), int(y1)), (int(x2), int(y2)), color, thickness)


def draw_lines(img, lines, color=[255, 0, 0], thickness=10):
    """
    NOTE: this is the function you might want to use as a starting point once you want to 
    average/extrapolate the line segments you detect to map out the full
    extent of the lane (going from the result shown in raw-lines-example.mp4
    to that shown in P1_example.mp4).  
    
    Think about things like separating line segments by their 
    slope ((y2-y1)/(x2-x1)) to decide which segments are part of the left
    line vs. the right line.  Then, you can average the position of each of 
    the lines and extrapolate to the top and bottom of the lane.
    
    This function draws `lines` with `color` and `thickness`.    
    Lines are drawn on the image inplace (mutates the image).
    If you want to make the lines semi-transparent, think about combining
    this function with the weighted_img() function below
    """
    
    # not wise to use this because car centre maybe offset from lane centre (only works for this proj)
    lane_center = img.shape[1]/2 ## approx lane center 
    
    leftX = []
    leftY = []
    
    rightX = []
    rightY = []
    
    
    for line in lines:
        for x1,y1,x2,y2 in line:
            
            # avoid dealing with infinite slope
            if x1 == x2:
                continue
            
            slope = (y2-y1)/(x2-x1)
            
            # slope is -ve for left lines as origin is at top-left in images
            # second condition rejects any lines on wrong side
            if (slope < 0) and x1 < lane_center and x2 < lane_center:
                leftX.extend([x1, x2])
                leftY.extend([y1, y2])
            
            if (slope > 0) and x1 > lane_center and x2 > lane_center:
                rightX.extend([x1, x2])
                rightY.extend([y1, y2])
            
            
    draw_helper(img, leftX, leftY, color, thickness)
    draw_helper(img, rightX, rightY, color, thickness)
            

def hough_lines(img, rho, theta, threshold, min_line_len, max_line_gap):
    """
    Applies probabilistic hough transform on the input image.
    Refer - https://stackoverflow.com/questions/40531468/explanation-of-rho-and-theta-parameters-in-houghlines
    
    Args:
        img: Input image (should be an edge img)
        rho (pixels): hough accumulator parameter rho
        theta (radians): hough accumulator parameter theta
        threshold: specifies the minimum number of votes (intersections in a given grid cell) 
            a candidate line needs to have to make it into the output
        min_line_len (pixels): min length of line. Lines shorter than this are rejected
        max_line_gap (pixels): max allowed gap between line segments to treat them as single line 
        
    Returns:
        input image with hough lines drawn.
    """
    hough_lines = cv2.HoughLinesP(img, rho, theta, threshold, np.array([]), minLineLength=min_line_len, maxLineGap=max_line_gap)
    line_img = np.zeros((img.shape[0], img.shape[1], 3), dtype=np.uint8)
    draw_lines(line_img, hough_lines)
    
    return hough_lines, line_img


# Python 3 has support for cool math symbols.
def weighted_img(line_img, initial_img, =0.8, =1., =0.):
    """
    `line_img` is the output of the hough_lines(), An image with lines drawn on it.
    Should be a blank image (all black) with lines drawn on it.
    
    `initial_img` should be the image before any processing.
    
    The result image is computed as follows:
    
    initial_img *  + img *  + 
    NOTE: initial_img and img must be the same shape!
    """
    return cv2.addWeighted(initial_img, , line_img, , )

def lane_detector(image, debug=False):
    
    if debug:
        plt.figure()
        fig, axes = plt.subplots(4,2, figsize=(15,18))
        axes[0][0].set_title('Original Image')
        axes[0][0].imshow(image)
    
    
    # Convert to grayscale
    gray = grayscale(image)
    if debug:
        axes[0][1].set_title('Grayscale Image')
        axes[0][1].imshow(gray, cmap='gray')
    
    
    # Smooth out noise from the grayscale image using gaussian blur
    gaussian_kernel = 5
    smooth_gray = gaussian_blur(gray, gaussian_kernel)
    if debug:
        axes[1][0].set_title('Smoothed image')
        axes[1][0].imshow(smooth_gray, cmap='gray')
        
        
    # Use Canny edge detector to detect edges
    # Recommended ratio between low and high threshold is 1:2 or 1:3 (by John F. Canny)
    low_threshold = 60
    high_threshold = 180
    edge = canny(smooth_gray, low_threshold, high_threshold)
    if debug:
        axes[1][1].set_title('Edges detected')
        axes[1][1].imshow(edge, cmap='gray')
        
        
    # Apply ROI mask 
    # Note cv2.fillpoly expects numpy array of arrays for the vertices argument
    ## Image shape is in (y,x) format i.e. height by width but vertices should be passed in (x,y) format
    vertices = [[(0, image.shape[0]), (460, 320), (520, 320), (image.shape[1], image.shape[0])]]
    fillpoly_vertices = np.array(vertices)
    masked = region_of_interest(edge, fillpoly_vertices)
    if debug:
        axes[2][0].set_title('Masked Image')
        axes[2][0].imshow(masked, cmap='gray')
        
        
    # Apply Probabilistic Hough Transform on the masked image
    rho = 2
    theta = np.pi/180
    threshold = 10        ## min votes required in accumulator to detect a line
    min_line_len = 40
    max_line_gap = 20
    hough_transform_lines, lane_img = hough_lines(masked, rho, theta, threshold, min_line_len, max_line_gap)
    if debug:
        hough_img = np.zeros_like(image)
        for line in hough_transform_lines:
            for x1, y1, x2, y2 in line:
                cv2.line(hough_img, (x1, y1), (x2, y2), (255, 0, 0), 10)
        # plot disjount houghlines
        # plot clean line image
        axes[2][1].set_title('Hough lines')
        axes[2][1].imshow(hough_img, cmap='gray')
        
        axes[3][0].set_title('Smooth lane line')
        axes[3][0].imshow(lane_img, cmap='gray')
        
    
    # Merge original image with hough lines
    
    final_img = weighted_img(lane_img, image)
    if debug:
        axes[3][1].set_title('Final Lane lines')
        axes[3][1].imshow(final_img)
        
        
    if not debug:
        return final_img
